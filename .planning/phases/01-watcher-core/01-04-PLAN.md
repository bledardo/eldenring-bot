```python
class ConsecutiveConfirmer:
    def __init__(self, required_count: int = 3):
        self.required = required_count
        self.count = 0

    def update(self, detected: bool) -> bool:
        if detected:
            self.count += 1
        else:
            self.count = 0
        return self.count >= self.required

    def reset(self):
        self.count = 0
```
